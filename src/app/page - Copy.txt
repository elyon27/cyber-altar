"use client";

import { FormEvent, useState } from "react";
import { useRouter } from "next/navigation";

export default function HomePage() {
  const router = useRouter();

  const [name, setName] = useState("");
  const [email, setEmail] = useState("");

  function handleEnter(e: FormEvent<HTMLFormElement>) {
    e.preventDefault();

    if (!name.trim()) {
      alert("Please enter your desired altar name.");
      return;
    }

    const params = new URLSearchParams({
      name: name.trim(),
    });

    if (email.trim()) {
      params.set("email", email.trim());
    }

    router.push(`/altar?${params.toString()}`);
  }

  return (
    <main className="min-h-screen bg-black text-white">
      <section className="mx-auto flex min-h-screen max-w-4xl items-center px-6 py-12">
        <div className="w-full rounded-3xl border border-white/10 bg-white/5 p-8 shadow-2xl">
          <div className="mb-8">
            <p className="mb-3 text-sm uppercase tracking-[0.3em] text-white/50">
              Cyber Altar
            </p>

            <h1 className="mb-4 text-4xl font-bold leading-tight md:text-5xl">
              Enter the Holy Place
            </h1>

            <p className="max-w-2xl text-base leading-7 text-white/75">
              Welcome, pilgrim. The Cyber Altar is a quiet place of prayer,
              reflection, and offering. Here you may enter a name by which you
              desire to be known upon the altar, and if you are new, you may
              also leave your email address as a sign of your pilgrimage. If you
              have already entered before, simply write your nickname or altar
              name and proceed into the prayer wall.
            </p>
          </div>

          <form onSubmit={handleEnter} className="space-y-5">
            <div>
              <label
                htmlFor="name"
                className="mb-2 block text-sm font-medium text-white/85"
              >
                Desired Name / Pilgrim Name
              </label>
              <input
                id="name"
                type="text"
                placeholder="Enter your desired altar name"
                value={name}
                onChange={(e) => setName(e.target.value)}
                className="w-full rounded-2xl border border-white/10 bg-black/30 px-4 py-3 text-white outline-none placeholder:text-white/35"
              />
            </div>

            <div>
              <label
                htmlFor="email"
                className="mb-2 block text-sm font-medium text-white/85"
              >
                Email Address
              </label>
              <input
                id="email"
                type="email"
                placeholder="For new pilgrims only (optional for returning users)"
                value={email}
                onChange={(e) => setEmail(e.target.value)}
                className="w-full rounded-2xl border border-white/10 bg-black/30 px-4 py-3 text-white outline-none placeholder:text-white/35"
              />
            </div>

            <div className="pt-2">
              <button
                type="submit"
                className="rounded-2xl border border-white/15 bg-white/10 px-6 py-3 text-sm font-semibold tracking-wide text-white transition hover:bg-white/15"
              >
                Enter the Holy Place
              </button>
            </div>
          </form>
        </div>
      </section>
    </main>
  );
}
