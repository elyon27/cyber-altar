'use client'

import { useEffect, useState } from 'react'
import { useRouter } from 'next/navigation'

type CyberAltarPageProps = {
  slug: string
}

type PrayerEntry = {
  id: string
  text: string
  createdAt: string
}

export default function CyberAltarPage({ slug }: CyberAltarPageProps) {
  const router = useRouter()

  const [prayer, setPrayer] = useState('')
  const [prayers, setPrayers] = useState<PrayerEntry[]>([])
  const [candleLit, setCandleLit] = useState(false)
  const [candleCount, setCandleCount] = useState(0)
  const [pilgrimName, setPilgrimName] = useState('Pilgrim')

  useEffect(() => {
    const storedPrayers = localStorage.getItem(`altar_prayers_${slug}`)
    const storedCandleCount = localStorage.getItem(`altar_candle_count_${slug}`)
    const storedPilgrimName =
      localStorage.getItem('cyber_altar_username') ||
      localStorage.getItem('altar_nickname') ||
      localStorage.getItem('pilgrimName')

    if (storedPilgrimName && storedPilgrimName.trim()) {
      setPilgrimName(storedPilgrimName.trim())
    }

    if (storedPrayers) {
      try {
        const parsed = JSON.parse(storedPrayers)

        const normalized: PrayerEntry[] = parsed.map((item: any, index: number) => {
          if (typeof item === 'string') {
            return {
              id: `legacy-${index}-${Date.now()}`,
              text: item,
              createdAt: new Date().toISOString(),
            }
          }

          return {
            id: item.id || `restored-${index}-${Date.now()}`,
            text: item.text || '',
            createdAt: item.createdAt || new Date().toISOString(),
          }
        })

        setPrayers(normalized)
        localStorage.setItem(`altar_prayers_${slug}`, JSON.stringify(normalized))
      } catch (error) {
        console.error('Failed to load stored prayers:', error)
        setPrayers([])
      }
    }

    if (storedCandleCount) {
      const parsedCount = Number(storedCandleCount)
      setCandleCount(Number.isNaN(parsedCount) ? 0 : parsedCount)
      setCandleLit(parsedCount > 0)
    }
  }, [slug])

  const submitPrayer = () => {
    const trimmedPrayer = prayer.trim()
    if (!trimmedPrayer) return

    const newPrayer: PrayerEntry = {
      id: crypto.randomUUID(),
      text: trimmedPrayer,
      createdAt: new Date().toISOString(),
    }

    const updatedPrayers = [newPrayer, ...prayers]
    setPrayers(updatedPrayers)
    localStorage.setItem(`altar_prayers_${slug}`, JSON.stringify(updatedPrayers))
    setPrayer('')
  }

  const lightCandle = () => {
    const updatedCount = candleCount + 1
    setCandleLit(true)
    setCandleCount(updatedCount)
    localStorage.setItem(`altar_candle_count_${slug}`, String(updatedCount))
  }

  const formatDate = (isoDate: string) => {
    return new Date(isoDate).toLocaleString('en-PH', {
      year: 'numeric',
      month: 'short',
      day: 'numeric',
      hour: 'numeric',
      minute: '2-digit',
    })
  }

  const goBackToAltarSelection = () => {
    router.push('/altar')
  }

  const exitToCrownMind = () => {
    window.location.href = 'https://crownmind.netlify.app'
  }

  return (
    <div
      style={{
        minHeight: '100vh',
        background: '#1e3a8a',
        color: 'white',
        padding: '40px 20px 80px',
        textAlign: 'center',
      }}
    >
      <h1 style={{ marginBottom: '8px' }}>The Holy Place</h1>
      <p style={{ marginTop: 0, fontSize: '18px', fontWeight: 600 }}>
        Pilgrim: {pilgrimName}
      </p>
      <p style={{ marginTop: '6px' }}>Altar: {slug}</p>

      <div
        style={{
          marginTop: '20px',
          display: 'flex',
          justifyContent: 'center',
          gap: '12px',
          flexWrap: 'wrap',
        }}
      >
        <button
          onClick={goBackToAltarSelection}
          style={{
            padding: '12px 20px',
            borderRadius: '8px',
            border: 'none',
            cursor: 'pointer',
            fontWeight: 600,
            fontSize: '15px',
          }}
        >
          Return to Altar Selection
        </button>

        <button
          onClick={exitToCrownMind}
          style={{
            padding: '12px 20px',
            borderRadius: '8px',
            border: 'none',
            cursor: 'pointer',
            fontWeight: 600,
            fontSize: '15px',
          }}
        >
          Exit to CrownMind
        </button>
      </div>

      <div
        style={{
          marginTop: '40px',
          display: 'flex',
          justifyContent: 'center',
          gap: '40px',
          alignItems: 'center',
          flexWrap: 'wrap',
        }}
      >
        <img
          src="/images/cross.png"
          alt="Cross"
          width="120"
          style={{ objectFit: 'contain' }}
        />

        <img
          src={`/altar/${slug}.jpg`}
          alt="Altar"
          width="260"
          style={{ objectFit: 'cover', borderRadius: '12px' }}
        />

        <div
          style={{
            display: 'flex',
            flexDirection: 'column',
            alignItems: 'center',
            gap: '10px',
          }}
        >
          <img
            src="/images/candle.jpg"
            alt="Candle"
            width="120"
            style={{
              opacity: candleLit ? 1 : 0.7,
              transition: '0.3s ease',
              objectFit: 'contain',
            }}
          />

          <div
            style={{
              fontSize: '14px',
              background: 'rgba(255,255,255,0.12)',
              padding: '8px 12px',
              borderRadius: '8px',
              minWidth: '140px',
            }}
          >
            Lit {candleCount} time{candleCount === 1 ? '' : 's'}
          </div>
        </div>
      </div>

      <div style={{ marginTop: '42px', display: 'flex', justifyContent: 'center' }}>
        <div
          style={{
            width: '100%',
            maxWidth: '520px',
            border: '2px solid rgba(255,255,255,0.45)',
            borderRadius: '12px',
            padding: '14px',
            background: 'rgba(255,255,255,0.06)',
            boxSizing: 'border-box',
          }}
        >
          <label
            htmlFor="prayer-box"
            style={{
              display: 'block',
              textAlign: 'left',
              marginBottom: '10px',
              fontSize: '16px',
              fontWeight: 600,
            }}
          >
            Enter your prayer
          </label>

          <textarea
            id="prayer-box"
            value={prayer}
            onChange={(e) => setPrayer(e.target.value)}
            placeholder="Write your prayer here..."
            style={{
              width: '100%',
              minHeight: '140px',
              padding: '12px',
              borderRadius: '8px',
              border: '1px solid rgba(255,255,255,0.25)',
              resize: 'vertical',
              boxSizing: 'border-box',
              fontSize: '15px',
            }}
          />

          <div
            style={{
              marginTop: '16px',
              display: 'flex',
              justifyContent: 'center',
              gap: '12px',
              flexWrap: 'wrap',
            }}
          >
            <button
              onClick={submitPrayer}
              style={{
                padding: '12px 22px',
                borderRadius: '8px',
                border: 'none',
                cursor: 'pointer',
                fontWeight: 600,
                fontSize: '15px',
              }}
            >
              Submit Prayer
            </button>

            <button
              onClick={lightCandle}
              style={{
                padding: '12px 22px',
                borderRadius: '8px',
                border: 'none',
                cursor: 'pointer',
                fontWeight: 600,
                fontSize: '15px',
              }}
            >
              Light Candle
            </button>
          </div>
        </div>
      </div>

      <div
        style={{
          marginTop: '42px',
          maxWidth: '720px',
          marginLeft: 'auto',
          marginRight: 'auto',
        }}
      >
        <h2 style={{ marginBottom: '18px' }}>Former Prayers</h2>

        {prayers.length === 0 ? (
          <p style={{ opacity: 0.85 }}>No prayers yet.</p>
        ) : (
          prayers.map((entry, index) => (
            <div
              key={entry.id || `${entry.createdAt}-${index}`}
              style={{
                background: 'rgba(255,255,255,0.1)',
                padding: '14px 18px',
                marginTop: '12px',
                borderRadius: '10px',
                textAlign: 'left',
              }}
            >
              <div
                style={{
                  fontSize: '13px',
                  opacity: 0.8,
                  marginBottom: '8px',
                }}
              >
                {formatDate(entry.createdAt)}
              </div>

              <div style={{ fontSize: '18px', lineHeight: 1.5 }}>
                {entry.text}
              </div>
            </div>
          ))
        )}
      </div>
    </div>
  )
}
